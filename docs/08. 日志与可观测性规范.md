# 日志与可观测性规范 (Logging & Observability)

> **核心原则**：系统在任何时刻的状态都必须可观察、可追踪、可重现。日志是系统的"黑匣子"，监控是系统的"心电图"。

## 一、后端日志规范（Python — structlog）

### 1.1 统一日志初始化

所有后端模块统一从 `backend/core/logger.py` 获取 logger，**禁止**使用 `print()` 或自行配置 logging。

```python
# backend/core/logger.py — 唯一初始化入口
import structlog
import logging
import sys
from backend.core.config import settings

def configure_logging():
    """在 FastAPI app 启动时调用一次"""
    log_level = logging.DEBUG if settings.DEBUG else logging.INFO

    structlog.configure(
        processors=[
            structlog.contextvars.merge_contextvars,     # 请求级上下文绑定
            structlog.stdlib.add_log_level,
            structlog.stdlib.add_logger_name,
            structlog.processors.TimeStamper(fmt="iso"),
            structlog.processors.StackInfoRenderer(),
            structlog.processors.format_exc_info,
            # 生产：JSON 输出；开发：彩色终端输出
            structlog.processors.JSONRenderer()
            if not settings.DEBUG
            else structlog.dev.ConsoleRenderer(),
        ],
        wrapper_class=structlog.make_filtering_bound_logger(log_level),
        context_class=dict,
        logger_factory=structlog.PrintLoggerFactory(sys.stdout),
    )

# 使用方式（各模块统一）
import structlog
logger = structlog.get_logger(__name__)
```

### 1.2 日志字段规范

每条日志必须包含以下字段，结构化输出方便机器解析：

```python
# 字段命名：snake_case，值用原始类型（非字符串化的对象）
logger.info(
    "order_submitted",              # event: 事件名，动词_名词格式
    symbol="0700.HK",              # 标的代码
    side="BUY",                    # 业务字段
    qty=100,                       # 数值类型，非字符串
    price=305.2,
    strategy="momentum_v3",
    latency_ms=8,                  # 延迟用 _ms 后缀（毫秒）
    trace_id="req-abc-123",        # 请求追踪 ID（可选）
)
```

**字段命名约定**：
| 后缀 | 含义 | 示例 |
|:---|:---|:---|
| `_ms` | 毫秒耗时 | `latency_ms=45` |
| `_count` | 计数 | `retry_count=3` |
| `_bytes` | 字节数 | `payload_bytes=1024` |
| `_url` | URL 地址 | `endpoint_url="/api/v1/..."` |
| `_id` | ID 类字段 | `order_id="ORD-001"` |

### 1.3 必须记录的关键事件

```python
# ✅ 系统启动/关闭
logger.info("service_started", host="0.0.0.0", port=8000, env="production")
logger.info("service_shutdown", reason="SIGTERM", graceful=True)

# ✅ 外部连接状态
logger.info("futu_connected", opend_host="127.0.0.1", opend_port=11111)
logger.error("futu_disconnected", retry_count=3, error=str(e))
logger.info("redis_connected", host=settings.REDIS_HOST, port=settings.REDIS_PORT)

# ✅ 关键业务事件
logger.info("order_submitted", symbol=symbol, side=side, qty=qty, mode="sandbox")
logger.warning("real_trade_blocked", reason="REAL_TRADE_EXECUTE=false")
logger.info("strategy_signal", strategy="momentum_v3", signal="BUY", symbol=symbol)

# ✅ 性能关键路径
logger.info("screener_completed", market="HK", conditions=3, results=127, elapsed_ms=340)
logger.warning("screener_slow", elapsed_ms=2100, threshold_ms=1000)

# ✅ 降级事件
logger.warning("datasource_degraded", from_source="futu", to_source="yfinance",
               reason="connection_timeout")
logger.warning("rate_limit_hit", source="yfinance", symbol=symbol, wait_seconds=60)

# ✅ 安全事件
logger.warning("api_key_missing", key_name="FUTU_API_KEY")
logger.error("unauthorized_access", path=request.url.path, ip=client_ip)
```

### 1.4 请求级上下文绑定（Trace ID）

```python
# backend/core/middleware.py
import uuid
import structlog

async def logging_middleware(request: Request, call_next):
    trace_id = request.headers.get("X-Trace-ID", str(uuid.uuid4())[:8])
    # 绑定到当前请求的上下文，后续所有 logger 调用自动携带
    structlog.contextvars.bind_contextvars(
        trace_id=trace_id,
        method=request.method,
        path=request.url.path,
    )
    response = await call_next(request)
    structlog.contextvars.clear_contextvars()
    return response
```

## 二、前端日志规范（TypeScript）

### 2.1 统一日志模块

```typescript
// frontend/src/lib/logger.ts — 统一入口，各模块 import 此文件

type LogLevel = 'debug' | 'info' | 'warn' | 'error'
type LogContext = Record<string, string | number | boolean | null>

interface Logger {
  debug(event: string, ctx?: LogContext): void
  info(event: string, ctx?: LogContext): void
  warn(event: string, ctx?: LogContext): void
  error(event: string, ctx?: LogContext): void
}

// 生产环境：info 以上才输出；错误上报到可观测平台（如 Sentry）
// 开发环境：全级别彩色输出

export const logger: Logger = {
  info: (event, ctx) => console.log(JSON.stringify({ level:'info', event, ...ctx, ts: Date.now() })),
  warn: (event, ctx) => console.warn(JSON.stringify({ level:'warn', event, ...ctx, ts: Date.now() })),
  error: (event, ctx) => {
    console.error(JSON.stringify({ level:'error', event, ...ctx, ts: Date.now() }))
    // 可接入 Sentry: Sentry.captureMessage(event, { extra: ctx })
  },
  debug: (event, ctx) => {
    if (import.meta.env.DEV) console.debug(event, ctx)
  },
}
```

### 2.2 前端关键事件记录

```typescript
// WebSocket 生命周期
logger.info('ws_connected', { url: wsUrl, symbol })
logger.warn('ws_disconnected', { code: event.code, reason: event.reason, reconnectIn: 2000 })
logger.error('ws_error', { message: error.message })

// 数据状态变化
logger.warn('data_stale', { symbol, staleSinceMs: Date.now() - lastUpdateTs })
logger.info('data_resumed', { symbol, gapMs: Date.now() - lastUpdateTs })

// 性能关键点
logger.info('chart_render', { symbol, dataPoints: klineData.length, renderMs: elapsed })
logger.warn('chart_slow', { symbol, renderMs: elapsed, thresholdMs: 16 })

// 用户关键操作（审计用）
logger.info('order_staged', { symbol, side, qty, price, mode: 'sandbox' })
logger.info('strategy_executed', { name: strategy.name, action: 'run_backtest' })
```

## 三、监控仪表盘规范（Prometheus + Grafana）

### 3.1 核心业务指标（Golden Signals）

```python
# backend/core/metrics.py — prometheus_client 指标定义

from prometheus_client import Counter, Histogram, Gauge

# 1. 流量指标
tick_received_total = Counter('tick_received_total', 'Tick 接收总数', ['symbol', 'source'])
api_requests_total = Counter('api_requests_total', '接口调用总数', ['method', 'path', 'status'])

# 2. 延迟指标
tick_latency_ms = Histogram('tick_latency_ms', 'Tick 端到端延迟(ms)',
                            buckets=[5, 10, 20, 50, 100, 200, 500])
api_latency_ms = Histogram('api_latency_ms', 'API 响应延迟(ms)',
                           ['path'], buckets=[10, 50, 100, 250, 500, 1000])
screener_latency_ms = Histogram('screener_latency_ms', '选股器执行延迟(ms)',
                                buckets=[100, 300, 500, 1000, 2000, 5000])

# 3. 错误率指标
futu_errors_total = Counter('futu_errors_total', 'Futu API 错误总数', ['error_type'])
llm_errors_total = Counter('llm_errors_total', 'LLM 调用错误总数', ['error_type'])

# 4. 饱和度指标
redis_memory_ratio = Gauge('redis_memory_ratio', 'Redis 内存使用率 0~1')
websocket_connections = Gauge('websocket_connections_active', '活跃 WebSocket 连接数')
task_queue_depth = Gauge('task_queue_depth', '后台任务队列深度')
```

### 3.2 Grafana 推荐看板布局

```
Row 1: 系统健康（红绿灯看板）
  Futu 连接状态 | Redis 连接状态 | PostgreSQL 连接状态 | WebSocket 活跃连接数

Row 2: 流量与延迟
  Tick 接收速率（Tick/s） | API 请求速率（req/s）
  Tick P99 延迟 | API P99 延迟 | 选股器 P99 延迟

Row 3: 错误率
  Futu 错误率（%） | LLM 错误率（%） | API 5xx 错误率（%）

Row 4: 饱和度
  Redis 内存使用率 | Python 进程 CPU | Python 进程 内存 RSS
```
